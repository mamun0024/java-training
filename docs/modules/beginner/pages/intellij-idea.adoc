=== Declaration

[source,java]
.psvm / main
----
    public static void main(String[] args) {
            //code
    }
----

=== Statement

[source,java]
.sout
----
        System.out.println(myVariable);
----

[source,java]
.ifn
----
        if (myVariable == null) {
            // code
        }
----

[source,java]
.inn
----
        if (myVariable != null) {
            //code
        }
----

[source,java]
.fori
----
        for (int i = 0; i < myVariable; i++) {
            //code
        }
----
